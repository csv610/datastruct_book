x = 23 -> x1 = 2, x0 = 3
y = 45 -> y1 = 4, y0 = 5

z2 = x1 * y1       = 2 * 4        = 8
z0 = x0 * y0       = 3 * 5        = 15
z1 = (x1+x0)*(y1+y0) - z2 - z0
    = (2+3) * (4+5) - 8 - 15
    = 5 * 9 - 8 - 15
    = 45 - 23 = 22

Result = z2*10^2 + z1*10^1 + z0
       = 8*100 + 22*10 + 15
       = 800 + 220 + 15
       = 1035
